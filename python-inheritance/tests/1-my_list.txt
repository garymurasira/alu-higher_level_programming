MyList = __import__('1-my_list').MyList

# Test 1: Append and print original list
my_list = MyList()

my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(5)

empty_list = MyList()

empty_list.print_sorted()

# Test 5: Negative numbers
neg_list = MyList([4, -1, 2, -3])

neg_list.print_sorted()
